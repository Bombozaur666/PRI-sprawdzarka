source symul_lib.tcl;
set towers 6
set neighbours(0) {5 1}
set neighbours(1) {0 2}
set neighbours(2) {1 3}
set neighbours(3) {2 4}
set neighbours(4) {3 5}
set neighbours(5) {4 0}

fiber create $towers
{
	set sum $id_los
	set count [expr $towers - 1]
	set result ""
	comm1_send $sum
	fiber yield;
	while {$run}
	{
		if {$comm0!="" && $result!="end"}
		{
			if {$licz>0}
			{
				set k $kom0
				set sum [expr $id_los + $k]
				set licz [expr $licz - 1]
				comm1_send $sum
			}
			else
			{
				set result "end"
			}
		}
		fiber yield;
	}
}

Init;

proc visualization {} {
	fiber_iterate {_puts "$id: $id_los, sum: $sum, $licz, $result"}
}

fiber yield; runda; visualization
